using Microsoft.Maui.Graphics;

namespace HyperPaint;

public sealed class DrawingDrawable : IDrawable
{
    public IReadOnlyList<Stroke> Strokes { get; set; } = Array.Empty<Stroke>();
    public Color Background { get; set; } = Color.FromArgb(\"#101014\");

    public void Draw(ICanvas canvas, RectF dirtyRect)
    {
        canvas.FillColor = Background;
        canvas.FillRectangle(dirtyRect);

        foreach (var s in Strokes)
        {
            if (s.Points.Count < 2) continue;

            if (s.IsEraser)
            {
                // Eraser: paint over with background
                canvas.StrokeColor = Background;
                canvas.StrokeSize = s.Thickness * 2f;
                canvas.StrokeLineCap = LineCap.Round;
            }
            else
            {
                canvas.StrokeColor = s.Color;
                canvas.StrokeSize = s.Thickness;
                canvas.StrokeLineCap = LineCap.Round;
            }

            var path = new PathF();
            path.MoveTo(s.Points[0]);
            for (int i = 1; i < s.Points.Count; i++)
                path.LineTo(s.Points[i]);

            canvas.DrawPath(path);
        }
    }
}
